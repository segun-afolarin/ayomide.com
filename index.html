PrivGuard – Code Implementation


# Python logic for email protection - PrivGuard

def is_suspicious(msg):
    keywords = ["free money", "click here", "urgent"]
    return any(k.lower() in msg.lower() for k in keywords)

def spam_protection(email):
    print(f"Starting protection for {email}...\n")
    sample_emails = [
        "Hello! Free money waiting for you!",
        "Team meeting notes",
        "Click here to claim your prize"
    ]
    for i, msg in enumerate(sample_emails, 1):
        status = "⚠ Suspicious" if is_suspicious(msg) else "✔ Safe"
        print(f"[{i}] {status}: {msg[:30]}...")
    print("\nEmail secured!")
    return "Email secured!"

# Example usage
spam_protection("user@example.com")
